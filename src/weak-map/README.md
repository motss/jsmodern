## Usage

```ts
import { extend } from 'jsmodern';
import {
  from,
  isWeakMap,
  of,
} from 'jsmodern/dist/weak-map.js';

extend({
  weakMap: [
    from,
    isWeakMap,
    of,
  ],
});

console.log([
  WeakMap.from,
  WeakMap.isWeakMap,
  WeakMap.of,
].every(n => 'function' === typeof(n)));
```
